cmake_minimum_required(VERSION 3.0)
project(navajo CXX)
set(CMAKE_SHARED_LIBRARY_LINK_CXX_FLAGS)
set(CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS} -std=c++11 -Wall -Wextra -Wpedantic -g3")
set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS} -std=c++11 -Wall -Wextra -Wpedantic -Werror -O3 -fno-unwind-tables -fno-asynchronous-unwind-tables -fno-exceptions -fno-rtti -fno-ident -Wl,--gc-sections,--strip-all,--build-id=none")
set(CMAKE_BUILD_TYPE Release)
file(GLOB HEADERS *.hpp)
file(GLOB SOURCES *.cpp)
add_executable(${PROJECT_NAME} ${HEADERS} ${SOURCES})
